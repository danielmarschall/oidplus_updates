			warn('Creating a backup as '.$backup_name);
			if (!@copy('res/OIDplus/privacy_documentation$dede.html', $backup_name)) {
				warn('Creation of backup failed');
			}
		}
		writefile_2();
		if (@sha1_file('res/OIDplus/privacy_documentation$dede.html') !== 'a7c85408ee8be3570f46aa17d5018f88c5c4c859') {
			warn('File cannot be written (checksum mismatch): res/OIDplus/privacy_documentation$dede.html');
		}
	}
}
