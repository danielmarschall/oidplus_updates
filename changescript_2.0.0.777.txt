			$backup_name = $tmp['dirname'].DIRECTORY_SEPARATOR.$tmp['filename'].'.'.date('Ymdhis',@filemtime('vendor/licenses')).(isset($tmp['extension']) ? '.'.$tmp['extension'] : '');
			warn('Creating a backup as '.$backup_name);
			if (!@copy('vendor/licenses', $backup_name)) {
				warn('Creation of backup failed');
			}
		}
		writefile_6();
		if (@sha1_file('vendor/licenses') !== 'e3b4bce8a097a61f2b4c8258cd63997725824271') {
			warn('File cannot be written (checksum mismatch): vendor/licenses');
		}
	}
}
